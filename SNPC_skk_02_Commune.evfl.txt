                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:002', false)

            else:

            System.EventFlags['cPlayer:SkkTalkTodayFlag'] = true

            branch:
                EventFlowSystemActor@UI.UIShopBuyAppear('cCommuneIslandShoes')
entrypoint Event14:

            elif not EventFlowSystemActor.ShoppingCapacity('cShopping', 65534):
                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:011', false)
                fork:
                    branch:
                        EventFlowSystemActor.UIMoneyDisappear()
                    branch:
                        EventFlowSystemActor@msg.MessageSuspend()
                    branch:
                        EventFlowSystemActor@UI.UIShopBuyDisappear()
            elif MainNpc.ShopHasBellCount(1.0):
                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:013', true)
                run Demo_Common::Demo_Cmn_Shopping()
                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:014', false)
                System.EventFlags['cPlayer:SkkPurchaseToday'] = true

            else:
                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:012', false)
                run skk_commune_Again_NoMoney()
        else:
            fork:
                branch:
                    EventFlowSystemActor.UIMoneyDisappear()
                branch:
                    EventFlowSystemActor@UI.UIShopBuyDisappear()
            if System.EventFlags['cPlayer:SkkPurchaseToday']:
                MainNpc.OpenMessageWindow('TalkSNpc/skk/SP_skk_02_Commune:010', false)
            else:
