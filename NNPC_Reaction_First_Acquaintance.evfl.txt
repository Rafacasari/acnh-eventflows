flow Root():
    switch MainNpc.NpcInducementType():
        case 0:
            if MainNpc.NpcBirthType('cForeign'):
                MainNpc.NpcSetLastLandWord()
                if MainNpc.EventFlags['cNpcMemory:SameLand']:
                    run A401()
                else:
                    run A411()
            else:
                run A421()
        case 1:
            MainNpc.NpcSetLastLandWord()
            if MainNpc.NpcIsInducementPlayer():
                run A001()
            elif MainNpc.InducePlayerExist():
                MainNpc.SetPlayerName(0)
                run A011()
            else:
                run A021()
        case 2:
            if MainNpc.NpcIsInducementPlayer():
                run A101()
            elif MainNpc.InducePlayerExist():
                MainNpc.SetPlayerName(0)
                run A111()
            else:
                run A121()
        case 3:
            if MainNpc.NpcIsInducementPlayer():
                run A201()
            elif MainNpc.InducePlayerExist():
                MainNpc.SetPlayerName(0)
                run A211()
            else:
                run A221()
        case 4:
            if MainNpc.NpcIsInducementPlayer():
                run A301()
            elif MainNpc.InducePlayerExist():
                MainNpc.SetPlayerName(0)
                run A311()
            else:
                run A321()
