flow RumorOP_Action_Villager():
    MainNpc.DecideRumorPlayer(0)
    MainNpc.SetPlayerName(3)
    if not MainNpc.NpcHasAcquaintanceshipRumorPlayer():
        MainNpc.OpenMessageWindow('TalkNNpc/B1_Bo/Free/BO_FreeD_RumorOP_Action:000', false)
        MainNpc.SetRumorDSkipInfo()
    elif MainNpc.CanSelectRumorMessage(0):
        switch MainNpc.IsRumorActionDaysAgo():
            case 0, 1:

                MainNpc.NpcOpenRumorMessage()
            case 2:
                MainNpc.OpenMessageWindow('TalkNNpc/B1_Bo/Free/BO_FreeD_RumorOP_Action:002', false)
                MainNpc.NpcOpenRumorMessage()
            case 3:
                MainNpc.OpenMessageWindow('TalkNNpc/B1_Bo/Free/BO_FreeD_RumorOP_Action:003', false)
                MainNpc.SetRumorDSkipInfo()
    else:
        switch MainNpc.IsRumorActionDaysAgo():
            case 0:
                MainNpc.OpenMessageWindow('TalkNNpc/B1_Bo/Free/BO_FreeD_RumorOP_Action:004', false)
                MainNpc.SetRumorDSkipInfo()
