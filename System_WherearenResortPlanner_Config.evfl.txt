flow EntryPoint_AuthorID():
    if System.EventFlags['cPlayer:WHEREAREN_DisclosePlannerID']:
        EventFlowSystemActor.OpenMessageWindow('TalkSys/SYS_PhoneApp:402', true)
        if EventFlowSystemActor.GeneralTalkChoice2() == 0:
            System.EventFlags['cPlayer:WHEREAREN_DisclosePlannerID'] = false
            EventFlowSystemActor.OpenMessageWindow('TalkSys/SYS_PhoneApp:405', true)
        else:
            run EntryPoint_Setting_Continue()
    else:
        EventFlowSystemActor.OpenMessageWindow('TalkSys/SYS_PhoneApp:403', true)
        if EventFlowSystemActor.GeneralTalkChoice2() == 0:
            System.EventFlags['cPlayer:WHEREAREN_DisclosePlannerID'] = true
            EventFlowSystemActor.OpenMessageWindow('TalkSys/SYS_PhoneApp:406', true)
        else:
            run EntryPoint_Setting_Continue()
    run EntryPoint_Repetition()

flow EntryPoint_Repetition():
    EventFlowSystemActor.OpenMessageWindow('TalkSys/SYS_PhoneApp:411', true)
    switch EventFlowSystemActor.GeneralTalkChoice3():

            run EntryPoint_AuthorID()
        case 1:
            run EntryPoint_PickupSet()
        case 2:
            run EntryPoint_Cancel()

flow EntryPoint_Setting_Continue():
